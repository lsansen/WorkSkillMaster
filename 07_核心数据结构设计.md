# 职场技能包大师 - 核心数据结构设计

## 技能包相关

### 1. 技能包（SkillPackage）
```typescript
interface SkillPackage {
  id: string;
  name: string;
  description: string;
  category: 'general' | 'professional';
  version: string;
  author: string;
  createdAt: string; // ISO date
  updatedAt: string; // ISO date
  parameters: SkillParameter[];
  template: string; // Markdown template
  modelRequirements: ModelRequirement[];
  tags: string[];
  enabled: boolean;
}
```

### 2. 技能包参数（SkillParameter）
```typescript
interface SkillParameter {
  id: string;
  name: string;
  type: 'text' | 'textarea' | 'file' | 'select' | 'checkbox';
  label: string;
  description: string;
  required: boolean;
  default?: string;
  options?: string[];
  placeholder?: string;
  fileTypes?: string[];
}
```

### 3. 模型需求（ModelRequirement）
```typescript
interface ModelRequirement {
  type: 'local' | 'cloud';
  modelType: 'text' | 'audio';
  minTokens?: number;
  recommendedModels?: string[];
  fallbackEnabled: boolean;
}
```

## AI模型相关

### 4. 模型（Model）
```typescript
interface Model {
  id: string;
  name: string;
  type: 'local' | 'cloud';
  provider: 'ollama' | 'openai' | 'anthropic' | 'deepseek' | 'qwen' | 'whisper';
  modelType: 'text' | 'audio';
  version: string;
  description: string;
  enabled: boolean;
  config: ModelConfig;
  tokenCost?: TokenCost;
}
```

### 5. 模型配置（ModelConfig）
```typescript
interface ModelConfig {
  baseUrl?: string; // 本地模型地址或API endpoint
  apiKey?: string; // 云端API密钥（加密存储）
  temperature: number;
  maxTokens: number;
  topP: number;
  frequencyPenalty: number;
  presencePenalty: number;
  timeout: number; // 超时时间（毫秒）
  maxRetries: number;
  localPath?: string; // 本地模型路径
}
```

### 6. Token成本（TokenCost）
```typescript
interface TokenCost {
  prompt: number; // 每1000 token的成本（美元）
  completion: number; // 每1000 token的成本（美元）
  currency: 'USD' | 'CNY';
  updatedAt: string; // ISO date
}
```

### 7. 模型路由规则（ModelRouteRule）
```typescript
interface ModelRouteRule {
  id: string;
  skillPackageId?: string;
  taskType: string; // 任务类型
  priority: number;
  modelIds: string[]; // 按优先级排序的模型ID
  fallbackStrategy: 'local' | 'cloud' | 'none';
  enabled: boolean;
}
```
