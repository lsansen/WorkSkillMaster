# 职场技能包大师 - 数据库设计

## 数据库概述

本系统使用SQLite作为本地数据库，用于存储系统配置、技能包信息、模型配置、历史记录等数据。SQLite具有轻量级、无需服务端、易于部署等特点，非常适合本地应用。

## 表结构设计

| 字段名 | 数据类型 | 约束 | 描述 |
|-------|---------|------|------|

| `name` | `TEXT` | `NOT NULL` | 技能包名称 |
| `description` | `TEXT` | `NOT NULL` | 技能包描述 |

| `author` | `TEXT` | `NOT NULL` | 技能包作者 |

## 数据库初始化SQL

```sql
-- 创建技能包表
CREATE TABLE IF NOT EXISTS skill_packages (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    description TEXT NOT NULL,
    category TEXT NOT NULL,
    version TEXT NOT NULL,
    author TEXT NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    template TEXT NOT NULL,
    enabled INTEGER NOT NULL DEFAULT 1
);

-- 创建技能包参数表
CREATE TABLE IF NOT EXISTS skill_package_parameters (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    skill_package_id TEXT NOT NULL,
    param_id TEXT NOT NULL,
    name TEXT NOT NULL,
    type TEXT NOT NULL,
    label TEXT NOT NULL,
    description TEXT NOT NULL,
    required INTEGER NOT NULL DEFAULT 0,
    default_value TEXT,
    placeholder TEXT,
    options TEXT,
    file_types TEXT,
    FOREIGN KEY (skill_package_id) REFERENCES skill_packages(id)
);

-- 创建模型需求表
CREATE TABLE IF NOT EXISTS model_requirements (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    skill_package_id TEXT NOT NULL,
    type TEXT NOT NULL,
    model_type TEXT NOT NULL,
    min_tokens INTEGER,
    recommended_models TEXT,
    fallback_enabled INTEGER NOT NULL DEFAULT 1,
    FOREIGN KEY (skill_package_id) REFERENCES skill_packages(id)
);

-- 创建技能包标签表
CREATE TABLE IF NOT EXISTS skill_package_tags (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    skill_package_id TEXT NOT NULL,
    tag TEXT NOT NULL,
    FOREIGN KEY (skill_package_id) REFERENCES skill_packages(id)
);

-- 创建模型表
CREATE TABLE IF NOT EXISTS models (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    type TEXT NOT NULL,
    provider TEXT NOT NULL,
    model_type TEXT NOT NULL,
    version TEXT NOT NULL,
    description TEXT NOT NULL,
    enabled INTEGER NOT NULL DEFAULT 1,
    config TEXT NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- 创建模型成本表
CREATE TABLE IF NOT EXISTS model_costs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    model_id TEXT NOT NULL,
    prompt_cost REAL NOT NULL,
    completion_cost REAL NOT NULL,
    currency TEXT NOT NULL DEFAULT 'USD',
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (model_id) REFERENCES models(id)
);

-- 创建模型路由规则表
CREATE TABLE IF NOT EXISTS model_routes (
    id TEXT PRIMARY KEY,
    skill_package_id TEXT,
    task_type TEXT NOT NULL,
    priority INTEGER NOT NULL DEFAULT 0,
    model_ids TEXT NOT NULL,
    fallback_strategy TEXT NOT NULL DEFAULT 'local',
    enabled INTEGER NOT NULL DEFAULT 1,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (skill_package_id) REFERENCES skill_packages(id)
);

-- 创建API密钥表
CREATE TABLE IF NOT EXISTS api_keys (
    id TEXT PRIMARY KEY,
    provider TEXT NOT NULL,
    name TEXT NOT NULL,
    encrypted_key TEXT NOT NULL,
    enabled INTEGER NOT NULL DEFAULT 1,
    usage_limit INTEGER,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    last_used_at TIMESTAMP
);

-- 创建用户配置表
CREATE TABLE IF NOT EXISTS user_config (
    id TEXT PRIMARY KEY,
    theme TEXT NOT NULL DEFAULT 'light',
    language TEXT NOT NULL DEFAULT 'zh-CN',
    default_model_id TEXT,
    auto_save INTEGER NOT NULL DEFAULT 1,
    notifications INTEGER NOT NULL DEFAULT 1,
    offline_mode INTEGER NOT NULL DEFAULT 0,
    budget_config TEXT NOT NULL,
    privacy_config TEXT NOT NULL,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (default_model_id) REFERENCES models(id)
);

-- 创建历史记录表
CREATE TABLE IF NOT EXISTS history (
    id TEXT PRIMARY KEY,
    skill_package_id TEXT NOT NULL,
    skill_package_name TEXT NOT NULL,
    input TEXT NOT NULL,
    output TEXT NOT NULL,
    model_id TEXT NOT NULL,
    model_name TEXT NOT NULL,
    prompt_tokens INTEGER NOT NULL,
    completion_tokens INTEGER NOT NULL,
    total_tokens INTEGER NOT NULL,
    cost REAL NOT NULL DEFAULT 0,
    duration INTEGER NOT NULL,
    favorited INTEGER NOT NULL DEFAULT 0,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (skill_package_id) REFERENCES skill_packages(id),
    FOREIGN KEY (model_id) REFERENCES models(id)
);

-- 创建历史记录标签表
CREATE TABLE IF NOT EXISTS history_tags (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    history_id TEXT NOT NULL,
    tag TEXT NOT NULL,
    FOREIGN KEY (history_id) REFERENCES history(id)
);

-- 创建任务表
CREATE TABLE IF NOT EXISTS tasks (
    id TEXT PRIMARY KEY,
    skill_package_id TEXT NOT NULL,
    input TEXT NOT NULL,
    status TEXT NOT NULL DEFAULT 'pending',
    progress INTEGER NOT NULL DEFAULT 0,
    result TEXT,
    error TEXT,
    model_id TEXT,
    prompt_tokens INTEGER,
    completion_tokens INTEGER,
    total_tokens INTEGER,
    cost REAL DEFAULT 0,
    started_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    completed_at TIMESTAMP,
    FOREIGN KEY (skill_package_id) REFERENCES skill_packages(id),
    FOREIGN KEY (model_id) REFERENCES models(id)
);

-- 创建文件表
CREATE TABLE IF NOT EXISTS files (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    type TEXT NOT NULL,
    size INTEGER NOT NULL,
    path TEXT NOT NULL,
    hash TEXT NOT NULL,
    metadata TEXT,
    content TEXT,
    processed INTEGER NOT NULL DEFAULT 0,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- 创建成本统计表
CREATE TABLE IF NOT EXISTS cost_statistics (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    date DATE NOT NULL,
    total_cost REAL NOT NULL DEFAULT 0,
    total_tokens INTEGER NOT NULL DEFAULT 0,
    model_breakdown TEXT NOT NULL,
    skill_package_breakdown TEXT NOT NULL
);

-- 创建索引
CREATE INDEX IF NOT EXISTS idx_skill_packages_category ON skill_packages(category);
CREATE INDEX IF NOT EXISTS idx_skill_packages_enabled ON skill_packages(enabled);
CREATE INDEX IF NOT EXISTS idx_skill_package_parameters_skill_package_id ON skill_package_parameters(skill_package_id);
CREATE INDEX IF NOT EXISTS idx_model_requirements_skill_package_id ON model_requirements(skill_package_id);
CREATE INDEX IF NOT EXISTS idx_skill_package_tags_skill_package_id ON skill_package_tags(skill_package_id);
CREATE INDEX IF NOT EXISTS idx_skill_package_tags_tag ON skill_package_tags(tag);
CREATE INDEX IF NOT EXISTS idx_models_type ON models(type);
CREATE INDEX IF NOT EXISTS idx_models_provider ON models(provider);
CREATE INDEX IF NOT EXISTS idx_models_model_type ON models(model_type);
CREATE INDEX IF NOT EXISTS idx_models_enabled ON models(enabled);
CREATE INDEX IF NOT EXISTS idx_model_costs_model_id ON model_costs(model_id);
CREATE INDEX IF NOT EXISTS idx_model_routes_skill_package_id ON model_routes(skill_package_id);
CREATE INDEX IF NOT EXISTS idx_model_routes_task_type ON model_routes(task_type);
CREATE INDEX IF NOT EXISTS idx_model_routes_priority ON model_routes(priority);
CREATE INDEX IF NOT EXISTS idx_api_keys_provider ON api_keys(provider);
CREATE INDEX IF NOT EXISTS idx_api_keys_enabled ON api_keys(enabled);
CREATE INDEX IF NOT EXISTS idx_history_skill_package_id ON history(skill_package_id);
CREATE INDEX IF NOT EXISTS idx_history_model_id ON history(model_id);
CREATE INDEX IF NOT EXISTS idx_history_created_at ON history(created_at);
CREATE INDEX IF NOT EXISTS idx_history_favorited ON history(favorited);
CREATE INDEX IF NOT EXISTS idx_history_tags_history_id ON history_tags(history_id);
CREATE INDEX IF NOT EXISTS idx_history_tags_tag ON history_tags(tag);
CREATE INDEX IF NOT EXISTS idx_tasks_skill_package_id ON tasks(skill_package_id);
CREATE INDEX IF NOT EXISTS idx_tasks_status ON tasks(status);
CREATE INDEX IF NOT EXISTS idx_tasks_started_at ON tasks(started_at);
CREATE INDEX IF NOT EXISTS idx_files_type ON files(type);
CREATE INDEX IF NOT EXISTS idx_files_processed ON files(processed);
CREATE INDEX IF NOT EXISTS idx_files_created_at ON files(created_at);
CREATE INDEX IF NOT EXISTS idx_cost_statistics_date ON cost_statistics(date);
```

## 数据安全考虑

1. **数据加密**：
   - API密钥使用AES-256加密存储
   - 敏感配置信息加密存储

2. **数据备份**：
   - 定期备份数据库
   - 支持手动导出配置

3. **数据清理**：
   - 自动清理过期数据
   - 支持手动清理历史记录

4. **访问控制**：
   - 本地数据库文件权限控制
   - 防止未授权访问

## 性能优化

1. **索引优化**：
   - 为常用查询字段创建索引
   - 定期优化索引

2. **查询优化**：
   - 使用参数化查询
   - 避免全表扫描

3. **缓存策略**：
   - 缓存常用数据
   - 减少数据库访问

4. **事务处理**：
   - 合理使用事务
   - 避免长事务
